Static polymorphism: Function overloading or operator overloading 

	class Dockyard {
		static void operator+(int n1, int n2);
		static void operator+(std::ofstream &outfile, n1, n1);
	}
Dynamic polymorphism: Function overriding or virtual function 
	  class Dockyard {
		public: void printContent() {
		std::cout << "5 ships\n";
          }

	  class Shipyard: public Dockyard{
		public: void printContent() {
		std::cout << "15 ships\n";
          }
