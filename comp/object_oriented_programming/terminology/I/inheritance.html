Classes gain statements from other classes 
	class Dockyard {
	public:
	static void printShips() {
		std::cout << "5 ships\n";
		}
		friend std::ostream operator<<(std::ostream &outfile, const Dockyard &rhs);
	}; 

	class Shipyard : public Dockard {
	public:
		static void printShips() {
			std::cout << "15 ships\n";
		}
	};
